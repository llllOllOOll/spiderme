Getting Started

Requires Zig 0.16.0-dev and libpq (for PostgreSQL)

Step 1: Fetch Spider

zig fetch --save git+https://github.com/llllOllOOll/spider

This updates your build.zig.zon with the dependency.

Step 2: Configure build.zig

const spider_dep = b.dependency("spider", .{
    .target = target,
});

const exe = b.addExecutable(.{
    .name = "spiderme",
    .root_module = b.createModule(.{
        .root_source_file = b.path("src/main.zig"),
        .target = target,
        .optimize = optimize,
        .link_libc = true, // required for libpq
        .imports = &.{
            .{ .name = "spider", .module = spider_dep.module("spider") },
        },
    }),
});

// Link against system-wide libpq
exe.root_module.linkSystemLibrary("pq", .{});

Step 3: Minimal Example

const std = @import("std");
const spider = @import("spider");

pub fn main(init: std.process.Init) !void {
    const allocator = init.gpa;
    const io = init.io;

    const server = try spider.Spider.init(allocator, io, "127.0.0.1", 8080);
    defer server.deinit();

    try server
        .get("/", pingHandler)
        .listen();
}

fn pingHandler(alc: std.mem.Allocator, _: *spider.Request) !spider.Response {
    return spider.Response.json(alc, .{ .msg = "pong" });
}

Step 4: Run

zig build run

Visit http://localhost:8080/

Expected response: {"msg":"pong"}

Router

Trie-based router supporting static routes, dynamic parameters (:id), and wildcards (*). Routes are chained directly on the server and dispatched per HTTP method. 

// Static route
server.get("/", homeHandler);

// Dynamic parameter — access via req.param("id")
server.get("/users/:id", userHandler);

// Wildcard — matches /assets/logo.png, /assets/css/main.css, etc.
server.get("/assets/*", staticHandler);

// Chained routes
server.get("/", homeHandler)
      .post("/users", createUser)
      .get("/users/:id", getUser)
      .listen() catch |err| return err;

fn userHandler(alc: std.mem.Allocator, req: *spider.Request) !spider.Response {
    const id = req.param("id") orelse return spider.Response.text(alc, "missing id");
    return spider.Response.json(alc, .{ .id = id });
}

WebSocket

Any path starting with /ws triggers WebSocket mode automatically. Spider manages the handshake, hub, and broadcast internally — no custom handler needed. Use spider.getWsHub().broadcast(msg) to push messages to all connected clients. 

// main.zig — initialize the hub once at startup
try spider.initWsHub(allocator, io);
defer spider.deinitWsHub(allocator);

// register the /ws route — Spider handles the rest
server.get("/ws", wsHandler);

// broadcast from any handler (e.g. on POST /messages)
const hub = spider.getWsHub();
const msg = try std.fmt.allocPrint(alc,
    "{\"user\":\"{s}\",\"text\":\"{s}\"}", .{ user, text });
hub.broadcast(msg);

// frontend — pure JS, no libraries needed
const ws = new WebSocket("ws://" + location.host + "/ws");
ws.onmessage = (e) => {
    const { user, text } = JSON.parse(e.data);
    appendMessage(user, text);
};

PostgreSQL

Built-in PostgreSQL client via spider.pg, wrapping libpq. Acquire a connection from the pool, run a query, iterate results by row and column index. 

const spg = @import("spider").pg;

pub fn findAll(self: DriverRepository) ![]Driver {
    const pool = db_pool.get();
    const conn = try pool.acquire();
    defer pool.release(conn);

    var result = spg.query(conn,
        "SELECT id, name, team, number FROM drivers ORDER BY number"
    ) catch |err| {
        std.log.err("query failed: {}", .{err});
        return err;
    };
    defer result.deinit();

    const count = result.rows();
    if (count == 0) return &[_]Driver{};

    const drivers = try self.allocator.alloc(Driver, count);
    for (drivers, 0..) |*driver, i| {
        driver.* = .{
            .id   = try std.fmt.parseInt(i32, result.getValue(i, 0), 10),
            .name = try self.allocator.dupe(u8, result.getValue(i, 1)),
        };
    }
    return drivers;
}

Logger

Spider outputs structured JSON logs for every request automatically. Use Zig's standard std.log for application-level logging — Spider pipes it through the same JSON format. 

// Application logging — standard Zig std.log
std.log.info("server started on port {d}", .{3000});
std.log.err("query failed: {}", .{err});

// {"level":"info","msg":"server_started","data":..port = 3000}
// [[SPIDER]] | 200 | 0ns | GET "/drivers"
// [[SPIDER]] | 404 | 0ns | GET "/not-found"

Pooling

Spider provides a connection pool for PostgreSQL via spider.pg.Pool. Initialize once at startup and share across all requests via a singleton pattern. 

// db/pool.zig — singleton
const spider_pg = @import("spider").pg;
var pool: ?*spider_pg.Pool = null;

pub fn init(p: *spider_pg.Pool) void {
    pool = p;
}

pub fn get() *spider_pg.Pool {
    return pool orelse @panic("DB pool not initialized");
}

// main.zig — initialize once at startup
var conn = try db.connect(allocator);
defer conn.deinit();
db_pool.init(&conn);

// any repository — acquire and release per request
const conn = try pool.acquire();
defer pool.release(conn);

Static Files

Serve static assets (images, CSS, JS) using a wildcard route and the Zig 0.16 file API. MIME type is detected automatically by file extension. 

// static_handler.zig
pub fn serve(alc: std.mem.Allocator, req: *spider.Request) !spider.Response {
    const path = req.path;
    const full_path = try std.fs.path.join(alc, &.{ ".", path });
    defer alc.free(full_path);

    var threaded = std.Io.Threaded.init_single_threaded;
    const io = threaded.io();

    const content = std.Io.Dir.cwd().readFileAlloc(
        io, full_path, alc, .limited(10 * 1024 * 1024),
    ) catch {
        return spider.Response.text(alc, "Not Found");
    };

    var res = spider.Response.init();
    res.allocator = alc;
    res.body = content;
    res.body_allocated = true;
    try res.headers.set(alc, "content-type", mimeType(path));
    return res;
}

// main.zig — register with wildcard
server.get("/assets/*", static.serve);

Graceful Shutdown

Spider handles graceful shutdown automatically — no extra configuration needed. Use Ctrl+\ to stop the server (sends SIGQUIT). The server finishes in-flight requests before exiting. 

// No setup needed — shutdown is handled automatically
server.get("/", homeHandler)
      .listen() catch |err| return err;

// defer ensures cleanup runs on shutdown
defer server.deinit();
defer conn.deinit();

// Stop the server:
// Ctrl+\ (SIGQUIT) — graceful shutdown

Group Routes

Group routes let you add a shared prefix to multiple routes — useful for versioned APIs like /api/v1. Two styles available: inline chaining with .groupGet / .groupPost, or a group object via app.group(prefix). 

// Style 1 — inline chaining
server.get("/", homeHandler)
      .groupGet("/api/v1", "/users", usersHandler)
      .groupGet("/api/v1", "/tasks", tasksHandler)
      .groupPost("/api/v1", "/tasks", createTaskHandler)
      .listen() catch |err| return err;

// Style 2 — group object
const api = app.group("/api/v1");
try api.get("/users", usersHandler);
try api.post("/tasks", createTaskHandler);

// Both register:  GET /api/v1/users
//                 GET /api/v1/tasks
//                 POST /api/v1/tasks

Request

Access route parameters, parse JSON bodies, and read form or query string values directly from *spider.Request. 

// Route parameter — /users/:id
const id = req.param("id") orelse return spider.Response.text(alc, "missing id");

// JSON body binding
const Task = struct { title: []const u8, done: bool };
const task = try req.bindJson(alc, Task);

// Form parameter (POST application/x-www-form-urlencoded)
const name = try req.formParam("name", alc);

// Query string (?page=2)
const page = try req.queryParam("page", alc);

// Raw path (e.g. "/users/42")
const path = req.path;

Response

Build responses with JSON, HTML, plain text, or redirects. Set custom status codes and headers as needed. 

// JSON
return spider.Response.json(alc, .{ .id = 1, .name = "Max" });

// HTML
return spider.Response.html(alc, "<h1>Hello</h1>");

// Plain text
return spider.Response.text(alc, "pong");

// Redirect
return spider.Response.redirect(alc, "/dashboard");

// Custom status + header
var res = try spider.Response.json(alc, task);
res.status = .created;
try res.headers.set(alc, "X-Request-Id", "abc123");
return res;

Middleware

Middleware functions run before every request handler. Chain them with server.use(fn). Each middleware receives the allocator, request, and a next function to continue the chain. Middleware is global — runs on every request. 

// MiddlewareFn type signature
pub const MiddlewareFn = *const fn(
    std.mem.Allocator,
    *spider.Request,
    spider.NextFn,
) anyerror!spider.Response;

// Built-in middleware (from spider.web)
const web = spider.web;

server.use(web.corsMiddleware)      // Adds CORS headers
    .use(web.loggerMiddleware)    // Logs: [GET] /path
    .use(web.authMiddleware)      // Checks x-api-key header
    .get("/", homeHandler)
    .listen() catch |err| return err;

// Custom middleware — call next() to continue
fn authMiddleware(
    alc: std.mem.Allocator,
    req: *spider.Request,
    next: spider.NextFn,
) !spider.Response {
    const key = req.header("x-api-key");
    if (key == null) {
        var res = try spider.Response.text(alc, "Unauthorized");
        res.status = .unauthorized;
        return res;
    }
    // Continue to next middleware or handler
    return next(alc, req);
}

// Middleware that modifies response
fn corsMiddleware(alc: std.mem.Allocator, req: *spider.Request, next: spider.NextFn) !spider.Response {
    var res = try next(alc, req);  // Get response from handler
    // Add headers to the response
    try res.headers.set(alc, "Access-Control-Allow-Origin", "*");
    try res.headers.set(alc, "Access-Control-Allow-Methods", "GET,POST,PUT,DELETE");
    return res;
}

Templates

Spider's template engine supports named blocks, variable interpolation, loops, conditionals, and includes. Templates are embedded at compile time via @embedFile. 

// docs_controller.zig — embed template and render blocks
const tmpl = @embedFile("views/index.html");

pub fn index(alc: std.mem.Allocator, _: *spider.Request) !spider.Response {
    const html = try spider.template.renderBlock(tmpl, "index", .{}, alc);
    return spider.Response.html(alc, html);
}

pub fn docRouter(alc: std.mem.Allocator, _: *spider.Request) !spider.Response {
    const html = try spider.template.renderBlock(tmpl, "doc_router", .{}, alc);
    return spider.Response.html(alc, html);
}

// Template with loop (e.g., listing drivers)
// Driver struct: { id: i32, name: []const u8, team: []const u8, number: i32 }
const html = try spider.template.renderBlock(tmpl, "drivers", .{
    .drivers = drivers,
}, alc);

Docker

Spider has an official Zig Docker image llllollooll/zig:master for fast multi-stage builds. 

# Dockerfile
FROM llllollooll/zig:master AS builder
WORKDIR /app
COPY . .
RUN apt-get update && apt-get install -y --no-install-recommends libpq-dev
RUN zig build -Doptimize=ReleaseSmall

FROM debian:bookworm-slim
RUN apt-get update && apt-get install -y --no-install-recommends libpq5 curl
WORKDIR /app
COPY --from=builder /app/zig-out/bin/spiderme /app/spiderme
EXPOSE 3000
CMD ["./spiderme"]
