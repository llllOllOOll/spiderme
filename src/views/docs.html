Quick Start

Spider is a web framework for Zig. This guide gets you from zero to a running HTTP server in under 5 minutes.

Prerequisites

You need Zig 0.16.0-dev.2676+ or later. If you don't have it yet, download it at ziglang.org/download.

Create a new project

sh
mkdir myapp && cd myapp
zig init

This generates build.zig, build.zig.zon, and src/main.zig.

Fetch Spider

sh
zig fetch --save git+https://github.com/llllOllOOll/spider

Spider is added to your build.zig.zon automatically:

zig
// build.zig.zon
.dependencies = .{
    .spider = .{
        .url = "git+https://github.com/llllOllOOll/spider#f79bfab69d1c99fa552583bd2cc627d34f894260",
        .hash = "spider-0.3.0-RIctlSneAQAo31OblO8nx0AZLGT0-d5qjb4aAjdzp4lP",
    },
},

Configure build.zig

Open build.zig and add Spider as a dependency of your executable:

zig
const spider_dep = b.dependency("spider", .{ .target = target });

const exe = b.addExecutable(.{
    .name = "myapp",
    .root_module = b.createModule(.{
        .root_source_file = b.path("src/main.zig"),
        .target = target,
        .optimize = optimize,
        .imports = &.{
            .{ .name = "spider", .module = spider_dep.module("spider") },
        },
    }),
});

Write your first handler

Replace src/main.zig with:

zig
const std = @import("std");
const spider = @import("spider");

pub fn main(init: std.process.Init) !void {
    const arena: std.mem.Allocator = init.arena.allocator();
    const io = init.io;
    const server = try spider.Spider.init(arena, io, "127.0.0.1", 8080);
    defer server.deinit();

    server.get("/ping", pingHandler).listen() catch |err| {
        std.log.err("server error: {}", .{err});
        return err;
    };
}

fn pingHandler(alc: std.mem.Allocator, _: *spider.Request) !spider.Response {
    return spider.Response.json(alc, .{ .msg = "pong" });
}

Run

sh
zig build run

Test it in another terminal:

sh
curl http://localhost:8080/ping
# {"msg":"pong"}
✓ You should also see Spider logging each request in the terminal:
[[SPIDER]] | 200 | 0ns | GET "/ping"

Router

Trie-based router supporting static routes, dynamic parameters (:id), and wildcards (*). Routes are chained directly on the server and dispatched per HTTP method.

zig
// Static route
server.get("/", homeHandler);

// Dynamic parameter — access via req.param("id")
server.get("/users/:id", userHandler);

// Wildcard — matches /assets/logo.png, /assets/css/main.css, etc.
server.get("/assets/*", staticHandler);

// Chained routes
server.get("/", homeHandler)
      .post("/users", createUser)
      .get("/users/:id", getUser)
      .listen() catch |err| return err;

fn userHandler(alc: std.mem.Allocator, req: *spider.Request) !spider.Response {
    const id = req.param("id") orelse return spider.Response.text(alc, "missing id");
    return spider.Response.json(alc, .{ .id = id });
}
{% end %} {% block "doc_request" %}

Request

Access route parameters, parse JSON bodies, and read form or query string values directly from *spider.Request.

zig
// Route parameter — /users/:id
const id = req.param("id") orelse return spider.Response.text(alc, "missing id");

// JSON body binding
const Task = struct { title: []const u8, done: bool };
const task = try req.bindJson(alc, Task);

// Form parameter (POST application/x-www-form-urlencoded)
const name = try req.formParam("name", alc);

// Query string (?page=2)
const page = try req.queryParam("page", alc);

// Raw path (e.g. "/users/42")
const path = req.path;
{% end %} {% block "doc_response" %}

Response

Build responses with JSON, HTML, plain text, or redirects. Set custom status codes and headers as needed.

zig
// JSON
return spider.Response.json(alc, .{ .id = 1, .name = "Max" });

// HTML
return spider.Response.html(alc, "<h1>Hello</h1>");

// Plain text
return spider.Response.text(alc, "pong");

// Redirect
return spider.Response.redirect(alc, "/dashboard");

// Custom status + header
var res = try spider.Response.json(alc, task);
res.status = .created;
try res.headers.set(alc, "X-Request-Id", "abc123");
return res;
{% end %} {% block "doc_middleware" %}

Middleware

Middleware functions run before every request handler. Chain them with server.use(fn). Each middleware receives the allocator, request, and a next function to continue the chain.

zig
// Built-in middleware
const web = spider.web;

server.use(web.corsMiddleware)
      .use(web.loggerMiddleware)
      .use(web.authMiddleware)
      .get("/", homeHandler)
      .listen() catch |err| return err;

// Custom middleware
fn authMiddleware(
    alc: std.mem.Allocator,
    req: *spider.Request,
    next: spider.NextFn,
) !spider.Response {
    const key = req.header("x-api-key");
    if (key == null) {
        var res = try spider.Response.text(alc, "Unauthorized");
        res.status = .unauthorized;
        return res;
    }
    return next(alc, req);
}
{% end %} {% block "doc_groups" %}

Group Routes

Group routes add a shared prefix to multiple routes — useful for versioned APIs like /api/v1.

zig
// Style 1 — inline chaining
server.get("/", homeHandler)
      .groupGet("/api/v1", "/users", usersHandler)
      .groupGet("/api/v1", "/tasks", tasksHandler)
      .groupPost("/api/v1", "/tasks", createTaskHandler)
      .listen() catch |err| return err;

// Style 2 — group object
const api = app.group("/api/v1");
try api.get("/users", usersHandler);
try api.post("/tasks", createTaskHandler);

PostgreSQL

Spider has a built-in PostgreSQL client via spider.pg. This guide goes from a fresh database to a working JSON endpoint, then shows how to move credentials to a .env file.

Prerequisites

The PostgreSQL client library (libpq) must be installed on your system.

On Arch Linux:

sh
pacman -S postgresql-libs

Verify the installation:

sh
pkg-config --exists libpq && echo "ok"

For other distributions (Debian, Ubuntu, Fedora, etc.) install the equivalent package — usually libpq-dev or postgresql-devel.

Start PostgreSQL

We'll use Docker to run PostgreSQL locally. Create a docker-compose.yml:

yaml
services:
  db:
    container_name: legends
    image: postgres:17
    environment:
      POSTGRES_USER: spider
      POSTGRES_PASSWORD: spider
      POSTGRES_DB: spider_db
    ports:
      - "5433:5432"
sh
docker compose up -d

Connect and create the table:

sh
docker exec -it legends psql -U spider -d spider_db
sql
CREATE TABLE legends (
    id         SERIAL PRIMARY KEY,
    name       VARCHAR(100) NOT NULL,
    band       VARCHAR(100) NOT NULL,
    instrument VARCHAR(50)  NOT NULL
);

INSERT INTO legends (name, band, instrument) VALUES
    ('Lemmy Kilmister', 'Motörhead',           'bass'),
    ('Ozzy Osbourne',   'Black Sabbath',        'vocals'),
    ('Tony Iommi',      'Black Sabbath',        'guitar'),
    ('Dio',             'Rainbow/Sabbath/Dio',  'vocals'),
    ('Rob Halford',     'Judas Priest',         'vocals');

Configure build.zig

Add link_libc and link against libpq:

zig
const exe = b.addExecutable(.{
    .name = "myapp",
    .root_module = b.createModule(.{
        .root_source_file = b.path("src/main.zig"),
        .target = target,
        .optimize = optimize,
        .link_libc = true, // required for libpq
        .imports = &.{
            .{ .name = "spider", .module = spider_dep.module("spider") },
        },
    }),
});
exe.root_module.linkSystemLibrary("pq", .{});

Connect and query

Define a struct that matches your table columns, connect to the database, and use mapAll to map the result automatically:

zig
const std = @import("std");
const spider = @import("spider");
const db = spider.pg;

const Legend = struct {
    id:         i32,
    name:       []const u8,
    band:       []const u8,
    instrument: []const u8,
};

fn legendsHandler(alc: std.mem.Allocator, _: *spider.Request) !spider.Response {
    var result = try db.query("SELECT id, name, band, instrument FROM legends");
    defer result.deinit();
    const legends = try result.mapAll(Legend, alc);
    return spider.Response.json(alc, legends);
}

pub fn main(init: std.process.Init) !void {
    const arena = init.arena.allocator();
    const io = init.io;

    try db.init(arena, .{
        .host     = "localhost",
        .port     = 5433,
        .user     = "spider",
        .password = "spider",
        .database = "spider_db",
    });
    defer db.deinit();

    const server = try spider.Spider.init(arena, io, "127.0.0.1", 8080);
    defer server.deinit();

    server.get("/legends", legendsHandler)
          .listen() catch |err| return err;
}
sh
curl http://localhost:8080/legends
# [{"id":1,"name":"Lemmy Kilmister","band":"Motörhead","instrument":"bass"},...]

Using .env

Hardcoding credentials is fine for development, but for real projects move them to a .env file. Spider reads it automatically with loadEnv — system environment variables always take priority.

sh
# .env
POSTGRES_HOST=localhost
POSTGRES_PORT=5433
POSTGRES_USER=spider
POSTGRES_PASSWORD=spider
POSTGRES_DB=spider_db

Then replace the hardcoded config with:

zig
spider.loadEnv(arena, ".env") catch {};  // silently skips if not found
try db.init(arena, .{});               // reads POSTGRES_* env vars
Note: Add .env to your .gitignore — never commit credentials.

Pooling

Spider manages the PostgreSQL connection pool internally via spider.pg. No manual acquire/release needed — query(), queryParams(), and exec() handle it automatically. The pool size defaults to 10.

zig
// Pool is invisible — Spider manages acquire/release internally
var result = try db.query("SELECT ...");   // acquires connection
defer result.deinit();                        // releases connection

// Configure pool size via DbConfig
try spg.init(allocator, .{
    .host      = "localhost",
    .database  = "spider_db",
    .pool_size = 20,   // default: 10
});

// Low-level API — for advanced use cases
const pool = spg.Pool;
const conn = try pool.acquire();
defer pool.release(conn);
var result = try spg.queryConn(conn, "SELECT ...");

WebSocket

Any path starting with /ws triggers WebSocket mode automatically. Spider manages the handshake, hub, and broadcast internally.

zig
// main.zig — initialize the hub once at startup
try spider.initWsHub(allocator, io);
defer spider.deinitWsHub(allocator);

server.get("/ws", wsHandler);

// broadcast from any handler
const hub = spider.getWsHub();
hub.broadcast(msg);

// frontend — pure JS
const ws = new WebSocket("ws://" + location.host + "/ws");
ws.onmessage = (e) => {
    const { user, text } = JSON.parse(e.data);
    appendMessage(user, text);
};

Templates

Spider's template engine is embedded at compile time via @embedFile. Use render for simple templates and renderBlock when your file contains named blocks.

Variable interpolation

Create src/views/index.html and use {{ }} to interpolate values:

html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Legends</title>
    <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/bulma@1.0.0/css/bulma.min.css">
</head>
<body>
<section class="section">
    <div class="container">
        <h1 class="title">{{ name }}</h1>
        <p class="subtitle">{{ band }}</p>
    </div>
</section>
</body>
</html>

In your handler, use spider.template.render and pass the data as a struct:

zig
const tmpl = @embedFile("views/index.html");

fn indexHandler(alc: std.mem.Allocator, _: *spider.Request) !spider.Response {
    const html = try spider.template.render(tmpl, .{
        .name = "Lemmy Kilmister",
        .band = "Motörhead",
    }, alc);
    return spider.Response.html(alc, html);
}

Loops

Use {% for %} to iterate over a slice. Here we fetch all legends from the database and render them in a table:

html
<tbody>
    {% for legend in legends %}
    <tr>
        <td>{{ legend.id }}</td>
        <td>{{ legend.name }}</td>
        <td>{{ legend.band }}</td>
        <td>{{ legend.instrument }}</td>
    </tr>
    {% endfor %}
</tbody>
zig
fn indexHandler(alc: std.mem.Allocator, _: *spider.Request) !spider.Response {
    var result = try db.query("SELECT id, name, band, instrument FROM legends");
    defer result.deinit();
    const legends = try result.mapAll(Legend, alc);
    const html = try spider.template.render(tmpl, .{ .legends = legends }, alc);
    return spider.Response.html(alc, html);
}

Named blocks

When your template file contains multiple named blocks, use renderBlock to render only one of them — useful for HTMX partial responses.

html
<!-- views/index.html -->
{% block "page" %}
<!DOCTYPE html>
...full page layout...
{% end %}

{% block "legends-table" %}
<tbody>
    {% for legend in legends %}
    <tr>
        <td>{{ legend.name }}</td>
        <td>{{ legend.band }}</td>
    </tr>
    {% endfor %}
</tbody>
{% end %}
zig
// full page
const html = try spider.template.renderBlock(tmpl, "page", .{ .legends = legends }, alc);

// partial — for HTMX swap
const html = try spider.template.renderBlock(tmpl, "legends-table", .{ .legends = legends }, alc);
render vs renderBlock — use render when the file is a single template. Use renderBlock when the file contains multiple named {% block %} sections.

Static Files

Serve static assets using a wildcard route. MIME type is detected automatically by file extension.

zig
pub fn serve(alc: std.mem.Allocator, req: *spider.Request) !spider.Response {
    const full_path = try std.fs.path.join(alc, &.{ ".", req.path });
    defer alc.free(full_path);

    var threaded = std.Io.Threaded.init_single_threaded;
    const io = threaded.io();
    const content = std.Io.Dir.cwd().readFileAlloc(
        io, full_path, alc, .limited(10 * 1024 * 1024),
    ) catch {
        return spider.Response.text(alc, "Not Found");
    };

    var res = spider.Response.init();
    res.body = content;
    res.body_allocated = true;
    try res.headers.set(alc, "content-type", mimeType(req.path));
    return res;
}

// main.zig
server.get("/assets/*", static.serve);

Logger

Spider outputs structured logs for every request automatically. Use Zig's standard std.log for application-level logging.

std.log.info("server started on port {d}", .{3000});

Graceful Shutdown

Spider handles graceful shutdown automatically. Use Ctrl+\ (SIGQUIT) to stop the server — in-flight requests finish before exit.

zig
// No setup needed — defer ensures cleanup on shutdown
defer server.deinit();
defer db.deinit();

server.get("/", homeHandler)
      .listen() catch |err| return err;

// Ctrl+\ → graceful shutdown

Docker

Spider has an official Zig Docker image llllollooll/zig:master for fast multi-stage builds.

dockerfile
FROM llllollooll/zig:master AS builder
WORKDIR /app
COPY . .
RUN apt-get update && apt-get install -y --no-install-recommends libpq-dev
RUN zig build -Doptimize=ReleaseSmall

FROM debian:bookworm-slim
RUN apt-get update && apt-get install -y --no-install-recommends libpq5 curl
WORKDIR /app
COPY --from=builder /app/zig-out/bin/spiderme /app/spiderme
EXPOSE 3000
CMD ["./spiderme"]
