const std = @import("std");
const spider = @import("spider");

pub fn main(init: std.process.Init) !void {
    var server = spider.app();
    defer server.deinit();

    server
        .get("/", homeHandler)
        .listen(.{}) catch {};
}

fn homeHandler(c: *spider.Ctx) !spider.Response {
    return c.json(.{ .msg = "pong" }, .{});
}
