HTTP Client

Spider includes a robust HTTP client for making external API requests, with HTTPS support and custom headers.

GET Requests

zig
const http_client = spider.http_client;

const response = try http_client.get(
allocator,
"https://jsonplaceholder.typicode.com/posts/1",
&.{
.{ .name = "User-Agent", .value = "Spider App" },
.{ .name = "Accept", .value = "application/json" },
}
);
defer allocator.free(response);

const Post = struct { title: []const u8, body: []const u8 };
const parsed = try std.json.parseFromSlice(Post, allocator, response, .{});
defer parsed.deinit();

POST Requests

zig
const body = try std.fmt.allocPrint(allocator,
"{{\"title\": \"New Post\", \"body\": \"Content\", \"userId\": 1}}", .{});
defer allocator.free(body);

const response = try http_client.post(
allocator,
"https://jsonplaceholder.typicode.com/posts",
body,
"application/json"
);
defer allocator.free(response);

Error Handling

zig
const HttpError = spider.http_client.HttpError;

const response = http_client.get(allocator, url, headers) catch |err| {
switch (err) {
HttpError.RequestFailed => {
std.log.err("Request to {s} failed", .{url});
return error.ExternalServiceUnavailable;
},
HttpError.BadStatus => {
std.log.err("Bad status from {s}", .{url});
return error.ExternalServiceError;
},
else => return err,
}
};
HTTPS Support: The HTTP client automatically handles HTTPS with proper certificate validation.
