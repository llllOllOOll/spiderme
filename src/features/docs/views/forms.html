FormData

Spider provides a powerful spider.form module for parsing form submissions with support for arrays, dot notation for nested fields, and automatic URL decoding.

Basic Usage

Parse form data from a POST request using req.form():

zig
fn handleSubmit(alc: std.mem.Allocator, req: *spider.Request) !spider.Response {
var form = try req.form(alc);
defer form.deinit();

const name = form.get("name");
const email = form.get("email");

return spider.Response.json(alc, .{ .name = name, .email = email });
}

Array Fields

Fields with [] suffix are parsed as arrays:

html
<form method="POST">
<input name="tags[]" value="zig">
<input name="tags[]" value="web">
<input name="tags[]" value="http">
</form>
zig
fn handleTags(alc: std.mem.Allocator, req: *spider.Request) !spider.Response {
var form = try req.form(alc);
defer form.deinit();

// Get all values as an array
const tags = form.getArray("tags");

// Returns ["zig", "web", "http"]
return spider.Response.json(alc, .{ .tags = tags });
}

Nested Fields (Dot Notation)

Access nested fields using dot notation:

html
<form method="POST">
<input name="user.name" value="John">
<input name="user.email" value="john@example.com">
<input name="user.roles[]" value="admin">
<input name="user.roles[]" value="editor">
</form>
zig
fn handleUser(alc: std.mem.Allocator, req: *spider.Request) !spider.Response {
var form = try req.form(alc);
defer form.deinit();

const name = form.get("user.name"); // "John"
const email = form.get("user.email"); // "john@example.com"
const roles = form.getArray("user.roles"); // ["admin", "editor"]

return spider.Response.json(alc, .{
.name = name,
.email = email,
.roles = roles,
});
}

Automatic URL Decoding

Form values are automatically URL decoded:

zig
// Input: "hello%20world" → Returns: "hello world"
// Input: "foo%2Fbar" → Returns: "foo/bar"
// Input: "hello+world" → Returns: "hello world" (plus = space)
const value = form.get("field");

Method Signature

zig
// req.form(allocator) returns spider.form.FormData
var form = try req.form(allocator);
defer form.deinit();

// Methods:
form.get(key) // ?[]const u8 - first value or null
form.getArray(key) // ?[]const []const u8 - all values or null
form.getNested(path) // ?[]const u8 - alias for get() with dot path
