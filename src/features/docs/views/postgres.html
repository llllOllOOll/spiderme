PostgreSQL

Spider has a built-in PostgreSQL client via spider.pg. This guide covers everything from basic queries to transactions with automatic logging.

Prerequisites

The PostgreSQL client library (libpq) must be installed on your system.

On Arch Linux:

pacman -S postgresql-libs

Verify the installation:

pkg-config --exists libpq && echo "ok"

For other distributions (Debian, Ubuntu, Fedora, etc.) install the equivalent package — usually libpq-dev or postgresql-devel.

Start PostgreSQL

We'll use Docker to run PostgreSQL locally. Create a docker-compose.yml:

services:
db:
container_name: spider_db
image: postgres:17
environment:
POSTGRES_USER: spider
POSTGRES_PASSWORD: spider
POSTGRES_DB: spider_db
ports:
- "5433:5432"
docker compose up -d

Connect and create a table:

docker exec -it spider_db psql -U spider -d spider_db
CREATE TABLE users (
id SERIAL PRIMARY KEY,
email VARCHAR(255) UNIQUE NOT NULL,
name VARCHAR(100) NOT NULL,
active BOOLEAN DEFAULT true
);

INSERT INTO users (email, name, active) VALUES
('alice@example.com', 'Alice Smith', true),
('bob@example.com', 'Bob Jones', false);

Configure build.zig

Add link_libc and link against libpq:

const exe = b.addExecutable(.{
.name = "myapp",
.root_module = b.createModule(.{
.root_source_file = b.path("src/main.zig"),
.target = target,
.optimize = optimize,
.link_libc = true, // required for libpq
.imports = &.{
.{ .name = "spider", .module = spider_dep.module("spider") },
},
}),
});
exe.root_module.linkSystemLibrary("pq", .{});

Connect and query

Define a struct that matches your table columns and use mapAll to map results automatically:

const db = spider.pg;

const User = struct {
id: i32,
email: []const u8,
name: []const u8,
active: bool,
};

fn usersHandler(alc: std.mem.Allocator, _: *spider.Request) !spider.Response {
var result = try db.query("SELECT id, email, name, active FROM users", .{});
defer result.deinit();
const users = try result.mapAll(User, alc);
return spider.Response.json(alc, users);
}

pub fn main(init: std.process.Init) !void {
const arena = init.arena.allocator();
const io = init.io;

try db.init(arena, io, .{});
defer db.deinit();

const server = try spider.Spider.init(arena, io, "127.0.0.1", 8080);
defer server.deinit();

server.get("/users", usersHandler)
.listen() catch |err| return err;
}

Query Parameters

Pass parameters using a struct literal with named fields. Spider handles type conversion automatically:

// From smoney/src/repositories/user_repository.zig
pub fn create(input: CreateUserInput) !User {
const sql = "INSERT INTO users (email, name) VALUES ($1, $2) RETURNING id, email, name";

return (try db.queryOneWith(User, sql, .{
input.email,
input.name,
})) orelse error.InsertFailed;
}

pub fn findById(id: i32) !?User {
const sql = "SELECT id, email, name FROM users WHERE id = $1";
return try db.queryOneWith(User, sql, .{id});
}

Supported parameter types: i32, i64, f32, f64, bool, []const u8 (strings), and optionals.

Named Column Access

Use result.get(row, "column_name") to access columns by name instead of index:

// From smoney/src/repositories/recurring_repository.zig
pub fn listByUser(self: Repository, user_id: i32) ![]Transaction {
var result = try db.query(
"SELECT id, user_id, title, amount FROM transactions WHERE user_id = $1",
.{user_id},
);
defer result.deinit();

var transactions = try self.allocator.alloc(Transaction, result.rows());
var i: usize = 0;
while (i < result.rows()) : (i += 1) {
// Access columns by name — more readable than index
transactions[i] = .{
.id = try std.fmt.parseInt(i32, result.get(i, "id"), 10),
.user_id = try std.fmt.parseInt(i32, result.get(i, "user_id"), 10),
.title = try self.allocator.dupe(u8, result.get(i, "title")),
.amount = try std.fmt.parseFloat(f64, result.get(i, "amount")),
};
}
return transactions;
}
Named vs Positional: Use result.get(row, "col") for readability. Use result.getValue(row, 0) for positional access (faster for known column order).

exec — INSERT/UPDATE/DELETE

Use db.exec() for mutations. It returns !void and handles connection management automatically:

// From smoney/src/repositories/recurring_repository.zig
pub fn updateNextDate(id: i32, next_date: []const u8) !void {
try db.exec(
"UPDATE recurring_transactions SET next_date = $1 WHERE id = $2",
.{ next_date, id },
);
}

pub fn toggleActive(id: i32, active: bool) !void {
try db.exec(
"UPDATE recurring_transactions SET active = $1 WHERE id = $2",
.{ active, id },
);
}

pub fn delete(id: i32) !void {
try db.exec(
"DELETE FROM recurring_transactions WHERE id = $1",
.{id},
);
}

Transactions

Use db.begin() to start a transaction. Always defer rollback() for automatic cleanup on error:

// From smoney/src/repositories/transaction_repository.zig
pub fn createBulk(inputs: []const CreateTransactionInput) !usize {
// Start transaction — defer rollback for automatic cleanup
var tx = try db.begin();
defer tx.rollback();

var inserted: usize = 0;
for (inputs) |input| {
// All queries share the same connection
try tx.exec(
"INSERT INTO transactions (user_id, date, title, amount) VALUES ($1, $2, $3, $4)",
.{ input.user_id, input.date, input.title, input.amount },
);
inserted += 1;
}

// Commit only if all inserts succeeded
try tx.commit();
return inserted;
}
Transaction pattern: defer tx.rollback() ensures the transaction is rolled back if any query fails. Only call commit() when everything succeeds.

Query Logging

Spider automatically logs every query with timing and parameters — no setup required:

// Output (INFO level): SQL, row count, execution time
pg: SELECT * FROM users WHERE id = $1 (1 rows, 123µs)

// Output (DEBUG level): SQL with parameter values
pg: $1 = "42"

The logging shows:

  • INFO level: SQL query, number of rows returned, execution time in microseconds
  • DEBUG level: Parameter values as readable $1 = "value" format

Controlling Log Level

Use LOG_LEVEL environment variable to control logging verbosity:

# Development — see all queries with params
export LOG_LEVEL=debug

# Production — hide pg logs, show only warnings/errors
export LOG_LEVEL=warn

Available levels: err, warn, info, debug

Manual Mapping

For complex type conversions, manually map rows using result.getValue() with positional access:

// From smoney/src/repositories/category_repository.zig
pub fn listByType(cat_type: CategoryType) ![]Category {
const type_str: []const u8 = switch (cat_type) {
.income => "income",
.expense => "expense",
.transfer => "transfer",
};

var result = try db.queryWith(
"SELECT id, name, icon, type FROM categories WHERE type = $1",
.{type_str},
);
defer result.deinit();

const count = result.rows();
var categories = try self.allocator.alloc(Category, count);
errdefer self.allocator.free(categories);

var i: usize = 0;
while (i < count) : (i += 1) {
const id = try std.fmt.parseInt(i32, result.getValue(i, 0), 10);
const name = result.getValue(i, 1);
const icon = result.getValue(i, 2);
const type_db = result.getValue(i, 3);

categories[i] = .{
.id = id,
.name = try self.allocator.dupe(u8, name),
.icon = try self.allocator.dupe(u8, icon),
.cat_type = if (std.mem.eql(u8, type_db, "income")) .income
else if (std.mem.eql(u8, type_db, "expense")) .expense
else .transfer,
};
}
return categories;
}

Using .env

Move credentials to a .env file for production. Spider reads them automatically:

# .env
POSTGRES_HOST=localhost
POSTGRES_PORT=5433
POSTGRES_USER=spider
POSTGRES_PASSWORD=spider
POSTGRES_DB=spider_db
spider.loadEnv(arena, ".env") catch {}; // silently skips if not found
try db.init(arena, io, .{}); // reads POSTGRES_* env vars
Note: Add .env to your .gitignore — never commit credentials.

API Reference

// Query methods
var result = try db.query(sql, .{}) // !Result - multiple rows
var row = try db.queryRow(sql, .{}) // !Result - single row
var user = try db.queryOneWith(User, sql, .{}) // !?User - single mapped
try db.exec(sql, .{}) // !void - INSERT/UPDATE/DELETE

// Transactions
var tx = try db.begin() // !Tx - start transaction
try tx.exec(sql, .{}) // !void - query in tx
try tx.commit() // !void - commit
tx.rollback() // void - safe to always call

// Result access
result.rows() // usize - row count
result.get(row, "column_name") // []const u8 - named
result.getValue(row, col) // []const u8 - positional
result.mapAll(User, alc) // ![]User - auto-map
result.mapOne(User, alc) // !?User - single mapped
