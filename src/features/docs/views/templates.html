Templates

Spider's template engine is embedded at compile time via @embedFile. Supports variable interpolation, conditionals, loops, named blocks, filters, and HTMX-aware rendering.

Variable interpolation

Use {{ }} to interpolate values:

html
{% raw %}
<h1>{{ title }}</h1>
<p>Welcome, {{ user_name }}!</p>
<p>Items: {{ count }}</p>
{% endraw %}
zig
const html = try spider.template.render(tmpl, .{
.title = "Dashboard",
.user_name = "Alice",
.count = 42,
}, alc);

Conditionals

Use {% if %} for conditional rendering:

html
{% raw %}
<!-- Basic condition -->
{% if is_admin %}
<p>Admin panel access</p>
{% endif %}

<!-- Negation -->
{% if !is_guest %}
<p>Welcome back!</p>
{% endif %}

<!-- Equality -->
{% if status == "active" %}
<span class="badge green">Active</span>
{% endif %}

<!-- Inequality -->
{% if role != "banned" %}
<a href="/profile">View Profile</a>
{% endif %}

<!-- Logical AND -->
{% if is_logged_in and is_premium %}
<p>Premium features unlocked!</p>
{% endif %}

<!-- Logical OR -->
{% if is_admin or is_moderator %}
<button>Moderate</button>
{% endif %}

<!-- Else branch -->
{% if has_permission %}
<a href="/edit">Edit</a>
{% else %}
<p>No edit access</p>
{% endif %}
{% endraw %}

Boolean strings "true", "1", "yes" evaluate to true. "false", "0", "no" evaluate to false.

Loops

Use {% for %} to iterate over a slice:

html
{% raw %}
<ul>
{% for item in items %}
<li>{{ item.name }} - {{ item.price }}</li>
{% endfor %}
</ul>
{% endraw %}
zig
const html = try spider.template.render(tmpl, .{
.items = items,
}, alc);

Named blocks & includes

Define blocks with {% block %} and include them with {% template %}:

html
{% raw %}
{% block "header" %}
<nav>Site Navigation</nav>
{% end %}

{% block "content" %}
<p>Page content here</p>
{% end %}
{% endraw %}
html
{% raw %}
<!-- In your layout -->
<header>{% template "header" %}</header>
<main>{% template "content" %}</main>
{% endraw %}
zig
// Render a specific block
const html = try spider.template.renderBlock(tmpl, "content", data, alc);

Filters

Apply filters with the | pipe operator:

html
{% raw %}
<p>Hello {{ name | default:"Guest" }}!</p>
<p>Value: {{ amount | default:"0.00" }}</p>
{% endraw %}
zig
// If name is empty/null, "Guest" is used instead
const html = try spider.template.render(tmpl, .{
.name = "", // or null
}, alc);

Rendering

Spider provides three rendering functions:

zig
// Simple template (no blocks)
const html = try spider.template.render(tmpl, data, alc);

// Render a specific named block
const html = try spider.template.renderBlock(tmpl, "content", data, alc);

// HTMX-aware rendering (automatic partial/full page)
return spider.renderView(alc, req, view_tmpl, data);

renderView — HTMX-Aware Rendering

renderView automatically handles HTMX requests:

  • HTMX request (HX-Request header): Renders only the "content" block
  • Direct request: Prepends registered layout, renders "base" block
zig
// main.zig — Register layout with AppConfig
const server = try spider.Spider.init(allocator, io, host, port, .{
.layout = @embedFile("views/layout.html"),
});

Template structure:

html
{% raw %}
<!-- views/layout.html -->
{% block "base" %}
<!DOCTYPE html>
<html>
<body>
<nav>Site Nav</nav>
<div id="content">
{% template "content" %}
</div>
</body>
</html>
{% end %}

<!-- views/dashboard.html -->
{% block "content" %}
<h1>Dashboard</h1>
<p>Welcome!</p>
{% end %}
{% endraw %}
zig
// Handler — renderView handles both cases automatically
fn dashboardHandler(alc: std.mem.Allocator, req: *spider.Request) !spider.Response {
const view = @embedFile("views/dashboard.html");
return spider.renderView(alc, req, view, .{});
}
render vs renderBlock vs renderView:
• render — simple templates without blocks
• renderBlock — render specific named block (HTMX partials)
• renderView — HTMX-aware with layout support
