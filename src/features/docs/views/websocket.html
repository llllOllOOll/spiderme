WebSocket

Any path starting with /ws triggers WebSocket mode automatically. Spider manages the handshake, hub, and broadcast internally.

zig
// main.zig — initialize the hub once at startup
try spider.initWsHub(allocator, io);
defer spider.deinitWsHub(allocator);

server.get("/ws", wsHandler);

// broadcast from any handler
const hub = spider.getWsHub();
hub.broadcast(msg);

// frontend — pure JS
const ws = new WebSocket("ws://" + location.host + "/ws");
ws.onmessage = (e) => {
const { user, text } = JSON.parse(e.data);
appendMessage(user, text);
};
